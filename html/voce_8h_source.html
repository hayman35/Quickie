89  jmethodID loadJavaMethodID(const std::string& functionName,
90  const std::string& functionSignature)
91  {
92  if (!gEnv)
93  {
94  log("ERROR", "Method ID cannot be found for function: "
95  + functionName + " before initialization.");
96  return 0;
97  }
98 
99  // Call the JNI function GetStaticMethodID, which performs a
100  // lookup for the given Java method. The lookup is based on
101  // the name of the method as well as the method signature. If
102  // the method does not exist, GetStaticMethodID
103  // returns zero (0). An immediate return from the native method
104  // at that point causes a NoSuchMethodError to be thrown in the
105  // Java application code. Note that the JNI uses the method
106  // signature to denote the return type of a Java method. The
107  // signature (I)V, for example, denotes a Java method that takes
108  // one argument of type int and has a return type void. The
109  // general form of a method signature argument is:
110  // "(argument-types)return-type".
111  jmethodID methodID = gEnv->GetStaticMethodID(gClass,
112  functionName.c_str(), functionSignature.c_str());
113 
114  if (0 == methodID)
115  {
116  log("ERROR", "Invalid Java method ID requested for: "
117  + functionName + ". Function call ignored.");
118  return 0;
119  }
120 
121  return methodID;
122  }

132  void init(const std::string& vocePath, bool initSynthesis,
133  bool initRecognition, const std::string& grammarPath,
134  const std::string& grammarName)
135  {
136  // Setup the Java virtual machine.
137  JavaVMInitArgs vm_args;
138  memset(&vm_args, 0, sizeof(vm_args));
139  vm_args.version = JNI_VERSION_1_4;
140 
141  long status = 0;
142 
143  // If recognition is being used, we need to increase the max heap
144  // size.
145  if (initRecognition)
146  {
147  vm_args.nOptions = 2;
148 
149  // Setup the VM options.
150  // TODO: check out other options to be used here, like disabling the
151  // JIT compiler.
152  JavaVMOption options[2];
153 
154  // Add the required Java class paths.
155  std::string classPathString = "-Djava.class.path=";
156  classPathString += vocePath;
157  classPathString += "/voce.jar";
158  char s[1024];
159  sprintf(s, classPathString.c_str());
160  options[0].optionString = s;
161  options[0].extraInfo = NULL;
162 
163  // Add an option to increase the max heap size.
164  options[1].optionString = "-Xmx256m";
165  options[1].extraInfo = NULL;
166  //options[1].optionString = "-Djava.compiler=NONE"; // Disable JIT.
167  //options[1].optionString = "-verbose:gc,class,jni";
168  vm_args.options = options;
169  //vm_args.ignoreUnrecognized = JNI_FALSE;
170 
171  // Create the VM.
172  status = JNI_CreateJavaVM(&internal::gJVM,
173  (void**)&internal::gEnv, &vm_args);
174  }
175  else
176  {
177  vm_args.nOptions = 1;
178 
179  // Setup the VM options.
180  // TODO: check out other options to be used here, like disabling the
181  // JIT compiler.
182  JavaVMOption options[1];
183 
184  // Add the required Java class paths.
185  std::string classPathString = "-Djava.class.path=";
186  classPathString += vocePath;
187  classPathString += "/voce.jar";
188  char s[1024];
189  sprintf(s, classPathString.c_str());
190  options[0].optionString = s;
191  options[0].extraInfo = NULL;
192  //options[1].optionString = "-Djava.compiler=NONE"; // Disable JIT.
193  //options[1].optionString = "-verbose:gc,class,jni";
194  vm_args.options = options;
195  //vm_args.ignoreUnrecognized = JNI_FALSE;
196 
197  // Create the VM.
198  status = JNI_CreateJavaVM(&internal::gJVM,
199  (void**)&internal::gEnv, &vm_args);
200  }
201 
202  if (status < 0)
203  {
204  internal::log("ERROR", "Java virtual machine cannot be created");
205  return;
206  }
207 
208  internal::log("", "Java virtual machine created");
209 
210  // Find the main Voce class by name.
211  jclass c = internal::gEnv->FindClass(internal::gClassName.c_str());
212 
213  if (0 == c)
214  {
215  internal::log("ERROR", "The requested Java class: "
216  + internal::gClassName + " could not be found. \
217 Make sure the class path correctly points to the Voce classes.");
218  return;
219  }
220 
221  // Store a global reference so the class won't go out of
222  // scope and get destroyed.
223  internal::gClass = (jclass)internal::gEnv->NewGlobalRef(c);
224 
225  // Cache the global method ID references for the various
226  // functions. This should save lookup time.
227  internal::gInitID = internal::loadJavaMethodID("init",
228  "(Ljava/lang/String;ZZLjava/lang/String;Ljava/lang/String;)V");
229  internal::gDestroyID = internal::loadJavaMethodID("destroy", "()V");
230  internal::gSynthesizeID = internal::loadJavaMethodID("synthesize",
231  "(Ljava/lang/String;)V");
232  internal::gStopSynthesizingID = internal::loadJavaMethodID(
233  "stopSynthesizing", "()V");
234  internal::gGetRecognizerQueueSizeID = internal::loadJavaMethodID(
235  "getRecognizerQueueSize", "()I");
236  internal::gPopRecognizedStringID = internal::loadJavaMethodID(
237  "popRecognizedString", "()Ljava/lang/String;");
238  internal::gSetRecognizerEnabledID = internal::loadJavaMethodID(
239  "setRecognizerEnabled", "(Z)V");
240  internal::gIsRecognizerEnabledID = internal::loadJavaMethodID(
241  "isRecognizerEnabled", "()Z");
242 
243  // Convert the C++ strings to Java strings.
244  jstring jStrVocePath = internal::gEnv->NewStringUTF(
245  vocePath.c_str());
246  jstring jStrGrammarPath = internal::gEnv->NewStringUTF(
247  grammarPath.c_str());
248  jstring jStrGrammarName = internal::gEnv->NewStringUTF(
249  grammarName.c_str());
250 
251  // Initialize the Java Voce stuff.
252  internal::gEnv->CallStaticVoidMethod(internal::gClass,
253  internal::gInitID, jStrVocePath, initSynthesis, initRecognition,
254  jStrGrammarPath, jStrGrammarName);
255  }
